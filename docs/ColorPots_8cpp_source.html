33 /**
34  * @brief Reads the average value (of n samples) for a potentiometer.
35  *
36  * The following function is used to read the average value of a potentiometer
37  * across n_avg samples.
38  *
39  * @param pin The pin of the potentiometer to read from.
40  * @param n_avg The number of samples to average over.
41  * @return uint8_t The average value of the potentiometer (0-255).
42  *
43  */
44 uint8_t avg_read_pot(uint8_t pin, uint8_t n_avg)
45 {
46  float sum = 0;
47 
48  for (unsigned int i = 0; i < n_avg; i++)
49  sum += analogRead(pin) >> SHFT_ADC_TO_UINT8;
50 
51  uint8_t ret = (uint8_t)round(sum / n_avg);
52  if (ret >= POT_UPPER_BOUND)
53  return 255;
54 
55  if (ret <= POT_LOWER_BOUND)
56  return 0;
57 
58  return ret;
59 }

72 // See header file for documentation.
73 bool ColorPots::update()
74 {
75  bool ret = false;
76 
77  uint8_t _r, _g, _b;
78  _r = avg_read_pot(pin_r, AVG_ADC_SAMPLES);
79  _g = avg_read_pot(pin_g, AVG_ADC_SAMPLES);
80  _b = avg_read_pot(pin_b, AVG_ADC_SAMPLES);
81 
82  // Check if any of the values have changed.
83  if (_r != r || _g != g || _b != b) {
84  last_change_tstamp = millis();
85  ret = true;
86  }
87 
88  r = _r;
89  g = _g;
90  b = _b;
91 
92  return ret;
93 }
