65 /**
66  * @brief Main loop for the firmware.
67  *
68  * The following function is the main loop for the
69  * firmware. It is responsible for continuously
70  * handling the hardware at a high level (through
71  * the use of hardware abstraction libraries/classes),
72  * as well as handling when the screensaver should be
73  * activated or deactivated.
74  *
75  * The firmware has been written in a way where the
76  * rotary encoder has the highset priority. Only once
77  * the rotary encoder has "cooled down" (see SizeEncoder.ready()),
78  * any remaining hardware will be handled.
79  *
80  */
81 void loop()
82 {
83  bool changed = false;
84 
85  // Check if rotary encoder is "cooled down",
86  // then check if hardware inputs have changed.
87  if (size_enc->ready()) {
88  changed = (strip->get_n_leds() != size_enc->ready_pos()); // Check if encoder has changed
89  changed = changed || color_pots->update(); // Check if color pots have changed
90  }
91 
92  // Enable screensaver if color pots are set to 0, and the pot
93  // and rotary encoder remain unchanged for SHOW_SCREENSAVER_TIMEOUT_MS
94  // ms (See config.h).
95  if (color_pots->t_since_last_change() >= SHOW_SCREENSAVER_AFTER_MSECS &&
96  size_enc->t_since_last_change() >= SHOW_SCREENSAVER_AFTER_MSECS &&
97  color_pots->zeroed()) {
98  display->start_screensaver();
99 
100  // Display until hardware has changed
101  while(!color_pots->update() && !size_enc->update())
102  display->update();
103 
104  display->stop_screensaver();
105  changed = true;
106  }
107 
108  // Handle hardware changes
109  if (changed) {
110  uint8_t r,g,b;
111  color_pots->get_rgb(r, g, b); // Get color from color pots
112  display->set_rgb(r, g, b); // Update color values on display
113  display->set_n_leds(size_enc->ready_pos()); // Update LED count on display
114  strip->set_rgb(r, g, b); // Update color on strip
115  strip->set_n_leds(size_enc->ready_pos()); // Update LED count on strip
116  display->update(); // Update display
117  }
118 
119  // Handle Rotary Encoder
120  size_enc->update();
121 }
