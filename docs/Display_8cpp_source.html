33 /**
34  * @brief Converts a uint8_t to a fixed size string.
35  *
36  * The following function converts a uint8_t to a fixed size string.
37  * In other words, each number is represented with 3 digits, where
38  * zeroes are prepended if necessary (ex. 001, 015 etc.).
39  *
40  * @param n The number to convert.
41  * @return The converted number as a fixed size string.
42  *
43  */
44 String uint8_to_fixed_str(uint8_t n)
45 {
46  String ret = "";
47  if (n < 100)
48  ret += "0";
49  if (n < 10)
50  ret += "0";
51  ret += String(n);
52  return ret;
53 }

55 /**
56  * @brief Converts RGB values to a string.
57  *
58  * The following function converts RGB values to a string with
59  * the following format:
60  * "R:<red> G:<green> B:<blue>"
61  *
62  * @param r The red value.
63  * @param g The green value.
64  * @param b The blue value.
65  * @return The converted RGB values as a string.
66  *
67  */
68 String rgb_to_str(uint8_t r, uint8_t g, uint8_t b)
69 {
70  String ret = "";
71  ret += "R:" + uint8_to_fixed_str(r) + " ";
72  ret += "G:" + uint8_to_fixed_str(g) + " ";
73  ret += "B:" + uint8_to_fixed_str(b);
74  return ret;
75 }

99 // See header file for documentation.
100 void Display::screensaver()
101 {
102  static bool wait = false;
103  static unsigned long wait_until;
104 
105  if (wait) {
106  wait = millis() < wait_until;
107  return;
108  }
109 
110  display->clearDisplay();
111  display->setTextSize(1);
112  display->setTextColor(WHITE);
113 
114  // Draw credits at top of screen.
115  display->setCursor(0,0);
116  display->println(SCREEN_SAVER_CREDITS_MSG);
117 
118  // Draw Waddle Dee with open eyes
119  if (random(0, 100) <= 60) {
120  display->drawBitmap(0, 20, WaddleDeeOpen, 128, 64, WHITE);
121  wait_until = millis() + SCREEN_SAVER_MIN_EYES_OPEN_TIME;
122  }
123 
124  // Draw Waddle Dee with closed eyes
125  else {
126  display->drawBitmap(0, 20, WaddleDeeClosed, 128, 64, WHITE);
127  wait_until = millis() + SCREEN_SAVER_MIN_BLINK_TIME;
128  }
129 
130  wait = true;
131  display->display();
132 }
